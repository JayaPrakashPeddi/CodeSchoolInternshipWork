function twoDigits(num) {
  const ones = {
    1: "One",
    2: "Two",
    3: "Three",
    4: "Four",
    5: "Five",
    6: "Six",
    7: "Seven",
    8: "Eight",
    9: "Nine",
  };

  const teens = {
    11: "Eleven",
    12: "Twelve",
    13: "Thirteen",
    14: "Fourteen",
    15: "Fifteen",
    16: "Sixteen",
    17: "Seventeen",
    18: "Eighteen",
    19: "Nineteen",
  };

  const tens = {
    10: "Ten",
    20: "Twenty",
    30: "Thirty",
    40: "Forty",
    50: "Fifty",
    60: "Sixty",
    70: "Seventy",
    80: "Eighty",
    90: "Ninety",
  };

  if (num === 0) {
    return "";
  } else if (num < 10) {
    return ones[num];
  } else if (num > 10 && num < 20) {
    return teens[num];
  } else if (num % 10 === 0) {
    return tens[num];
  } else {
    return tens[num - (num % 10)] + " " + ones[num % 10];
  }
}
